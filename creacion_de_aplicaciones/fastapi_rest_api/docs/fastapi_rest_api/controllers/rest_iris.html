 30    def revert_mapping(y:int or str) -> str:
 31        """
 32        Método para revertir el mapeo del resultado de la predicción
 33
 34        :param y: Resultado de la predicción
 35        :type y: int
 36        :return: Nombre de la especie

 41                return "setosa"

 51    def predict_results(self) -> list:
 52        """
 53        Método para predecir los resultados de la clasificación
 54
 55        :return: Resultados de la clasificación
 56        :rtype: list
 57        """
 58        # Carga de datos
 59        X_train = self.df_train.drop(["id", 'species'], axis=1)
 60        self.df_train.species = self.df_train.species.map({'setosa': 0, 'versicolor': 1, 'virginica': 2})
 61        y_train = self.df_train["species"]
 62        
 63        # Comprobación de df_test, en caso de que no exista, se usa df_train y se divide en train y test
 64        if self.df_test is not None:
 65            X_train = X_train.values
 66            X_test = self.df_test.values
 67            y_train = y_train.values
 68            # Lanzamiento de predicciones
 69            if self.classifier.lower() == 'randomforest':
 70                results = self.__run_random_forest(X_train, X_test, y_train)
 71            else:
 72                results = self.__run_decision_tree(X_train, X_test, y_train)
 73        else:
 74            results = []
 75            X = self.df_train.values
 76            y = y_train.values
 77            X_train, X_test, y_train, y_test = train_test_split(X, y, test_size=0.2)
 78            # Lanzamiento de predicciones
 79            results.append(self.__run_decision_tree(X_train, X_test, y_train, y_test))
 80            results.append(self.__run_random_forest(X_train, X_test, y_train, y_test))
 81        return results

 83    def __run_decision_tree(self, X_train, X_test, y_train, y_test=None):
 84        """
 85        Método para lanzar la predicción de un clasificador de árboles de decisión
 86
 87        :param X_train: Datos de entrenamiento
 88        :type X_train: array
 89        :param X_test: Datos de prueba
 90        :type X_test: array
 91        :param y_train: Resultados de entrenamiento
 92        :type y_train: array
 93        :param y_test: Resultados de prueba
 94        :type y_test: array
 95        :return: Resultados de la predicción
 96        :rtype: list
 97        """
 98        clf = DecisionTreeClassifier(random_state=42)
 99        clf.fit(X_train, y_train)
100        y_pred = clf.predict(X_test)
101        # En caso de no contar con y_test, se devuelve el resultado de la predicción
102        if y_test is not None:
103            score = f"{trunc(clf.score(X_test, y_test)*100)}%"
104            return y_test.tolist(), y_pred.tolist(), score, "Decision Tree"
105        return self.revert_mapping(y_pred.tolist()[0]), "Decission Tree Classifier"

177        response = {}
178        # Es necesario leer el dataset cada vez que se quiera obtener la precisión
179        iris_ml = IrisML(pd.read_csv(self.iris_path))
180        response["species"] = self.iris_df.species.value_counts().to_dict()
181        # Obtención de los resultados de las predicciones
182        results = iris_ml.predict_results()
183        response["algorithms"] = []
184        for result in results:
185            y_test, y_pred, score, classifier = result
186            tmp = {"name": classifier, "results": {

191            response["algorithms"].append(tmp)

215    def post_predict(self, data:dict) -> dict:
216        """
217        Método para predecir los datos de prueba
218
219        :param test_df: Datos de prueba
220        :type test_df: pd.DataFrame
221        :return: Resultados de la predicción
222        :rtype: dict
223        """
224        test_df = pd.DataFrame({
225            "sepal_length": [data["sepal_length"]],
226            "sepal_width": [data["sepal_width"]],
227            "petal_length": [data["petal_length"]],
228            "petal_width": [data["petal_width"]]
229        })
230        # Run prediction
231        iris_ml = IrisML(pd.read_csv(self.iris_path), test_df, data["classifier"])
232        predicted, classifier = iris_ml.predict_results()
233        # Store results in iris.csv
234        new_id = len(self.iris_df) + 1
235        self.iris_df = pd.concat([self.iris_df, pd.DataFrame({
236            "id": [new_id],
237            "sepal_length": [data["sepal_length"]],
238            "sepal_width": [data["sepal_width"]],
239            "petal_length": [data["petal_length"]],
240            "petal_width": [data["petal_width"]],
241            "species": [predicted],
242        }, index=[new_id])])
243        # Write to csv
244        self.iris_df["species"] = IrisML.revert_mapping(self.iris_df["species"])
245        self.update_csv()
246        data["id"] = new_id
247        data["predicted_species"] = predicted
248        data["predicted_model"] = classifier
249        return data

269        # Write to csv
270        self.update_csv()
